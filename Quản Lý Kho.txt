    <script>
        // Dữ liệu danh sách nguyên liệu y hệt trong ảnh của bạn
        const ingredientsData = [
            { name: "Cà phê hạt", unit: "kg", amount: 15, icon: "☕" },
            { name: "Sữa tươi", unit: "lít", amount: 48, icon: "🥛" },
            { name: "Đường", unit: "kg", amount: 30, icon: "🍚" },
            { name: "Bột cacao", unit: "kg", amount: 12, icon: "🍫" },
            { name: "Trà xanh", unit: "kg", amount: 8, icon: "🍃" },
            { name: "Syrup Vanilla", unit: "ml", amount: 750, icon: "🍾" },
            { name: "Whipping Cream", unit: "lít", amount: 20, icon: "🍦" }
        ];

        function renderIngredients() {
            const htmlContent = ingredientsData.map(item => `
                <div class="ingredient-card">
                    <div class="ingredient-icon">${item.icon}</div>
                    <div class="ingredient-info">
                        <div class="name">${item.name}</div>
                        <div class="unit">Đơn vị: ${item.unit}</div>
                    </div>
                    <div class="ingredient-stock">
                        <div class="amount">${item.amount}</div>
                        <div class="unit-label">${item.unit}</div>
                    </div>
                    <div class="actions">
                        <button class="btn-edit"><span class="material-icons-outlined" style="font-size: 18px;">edit</span></button>
                        <button class="btn-delete"><span class="material-icons-outlined" style="font-size: 18px;">delete</span></button>
                    </div>
                </div>
            `).join('');

            document.getElementById('list-container-1').innerHTML = htmlContent;
            document.getElementById('list-container-2').innerHTML = htmlContent;
        }

        function switchTab(screenId, element) {
            const screens = document.querySelectorAll('.screen');
            screens.forEach(screen => screen.classList.remove('active'));

            const navItems = document.querySelectorAll('.nav-item');
            navItems.forEach(item => item.classList.remove('active'));

            document.getElementById(screen-${screenId}).classList.add('active');
            element.classList.add('active');
        }

        renderIngredients();
    </script>
